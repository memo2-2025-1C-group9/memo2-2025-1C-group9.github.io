41@app.exception_handler(HTTPException) 

42async def http_exception_handler(request: Request, exc: HTTPException): 

43 # Determinar el título basado en el código de status 

44 title = "Error de Servidor" 

45 if exc.status_code == 401: 

46 title = "No Autorizado" 

47 elif exc.status_code == 403: 

48 title = "Prohibido" 

49 elif exc.status_code == 404: 

50 title = "No Encontrado" 

51 elif exc.status_code == 400: 

52 title = "Solicitud Incorrecta" 

53 elif exc.status_code == 422: 

54 title = "Error de Validación" 

55 elif exc.status_code < 500: 

56 title = "Error de Cliente" 

57 

58 logging.error( 

59 f"HTTPException manejada: {exc.detail} (status: {exc.status_code}, url: {request.url})" 

60 ) 

61 

62 headers = exc.headers or {} 

63 

64 headers["Content-Type"] = "application/problem+json" 

65 

66 headers["Access-Control-Allow-Origin"] = "*" 

67 

68 return problem_detail_response( 

69 status_code=exc.status_code, 

70 title=title, 

71 detail=exc.detail, 

72 instance=str(request.url), 

73 headers=headers, 

74 )

85@app.exception_handler(RequestValidationError) 

86async def validation_exception_handler(request: Request, exc: RequestValidationError): 

87 """Manejador para errores de validación de petición""" 

88 logging.error(f"Error de validación: {str(exc)}") 

89 logging.error(traceback.format_exc()) 

90 

91 error_details = [] 

92 for error in exc.errors(): 

93 loc = " -> ".join([str(x) for x in error.get("loc", [])]) 

94 msg = error.get("msg", "Error de validación") 

95 error_details.append(f"{loc}: {msg}") 

96 

97 detail = ( 

98 "; ".join(error_details) if error_details else "Error de validación de datos" 

99 ) 

100 

101 return problem_detail_response( 

102 status_code=422, 

103 title="Error de Validación", 

104 detail=detail, 

105 instance=str(request.url), 

106 headers={"Content-Type": "application/problem+json"}, 

107 )
